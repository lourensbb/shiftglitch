utils/payfast.js — paste this into every project

The signature calculation is where 90% of PayFast integrations break. Three rules: alphabetical key order, PHP-style URL encoding (space = + not %20), MD5 hash including the passphrase at the end.

const crypto = require("crypto");

/**
 * Generates a valid PayFast signature.
 * data       = { key: "value", ... }  — all values must be strings
 * passphrase = your PayFast passphrase (empty string if none set)
 */
function generateSignature(data, passphrase = "") {

  // Step 1: Sort keys alphabetically
  const sortedKeys = Object.keys(data).sort();

  // Step 2: Build query string, skip empty/null values
  const parts = sortedKeys
    .filter(k => data[k] !== "" && data[k] != null)
    .map(k => k + "=" + pfEncode(data[k]));

  let str = parts.join("&");

  // Step 3: Append passphrase if you have one
  if (passphrase) {
    str += "&passphrase=" + pfEncode(passphrase);
  }

  // Step 4: MD5 hash the whole string
  return crypto.createHash("md5").update(str).digest("hex");
}

// PayFast uses PHP urlencode style: space becomes + not %20
function pfEncode(val) {
  return encodeURIComponent(String(val)).replace(/%20/g, "+");
}

/**
 * Validates an incoming ITN notification from PayFast.
 * Call this inside your POST /payfast/notify route.
 */
async function validateITN(pfData, rawBody) {

  // Check the signature
  const { signature, ...rest } = pfData;
  const calculated = generateSignature(rest, process.env.PAYFAST_PASSPHRASE);
  if (calculated !== signature) {
    return { valid: false, reason: "Signature mismatch" };
  }

  // Ping PayFast to confirm the payment actually happened
  const host = process.env.PAYFAST_SANDBOX === "true"
    ? "sandbox.payfast.co.za"
    : "www.payfast.co.za";

  const res = await fetch("https://" + host + "/eng/query/validate", {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: rawBody,
  });

  const txt = await res.text();
  return txt === "VALID"
    ? { valid: true }
    : { valid: false, reason: "PayFast ping returned: " + txt };
}

module.exports = { generateSignature, pfEncode, validateITN };

Backend — create the payment (Node.js / Express)

// routes/payment.js
const express = require("express");
const router  = express.Router();
const { generateSignature } = require("../utils/payfast");

router.post("/create-payment", async (req, res) => {
  const { amount, itemName, userId, email, firstName, lastName } = req.body;

  const isSandbox = process.env.PAYFAST_SANDBOX === "true";
  const host      = isSandbox ? "sandbox.payfast.co.za" : "www.payfast.co.za";

  const data = {
    merchant_id:   process.env.PAYFAST_MERCHANT_ID,
    merchant_key:  process.env.PAYFAST_MERCHANT_KEY,
    return_url:    "https://yoursite.com/payment/success",
    cancel_url:    "https://yoursite.com/payment/cancel",
    notify_url:    "https://yoursite.com/api/payfast/notify",  // must be public!
    name_first:    firstName    || "",
    name_last:     lastName     || "",
    email_address: email        || "",
    m_payment_id:  "order_" + Date.now() + "_" + userId,  // your unique reference
    amount:        parseFloat(amount).toFixed(2),          // MUST be "100.00" format
    item_name:     itemName,
    custom_str1:   String(userId),   // store your userId here, gets echoed in ITN
  };

  // Remove empty fields before signing
  const clean = Object.fromEntries(
    Object.entries(data).filter(([, v]) => v !== "")
  );

  // Add the signature last
  clean.signature = generateSignature(clean, process.env.PAYFAST_PASSPHRASE);

  res.json({
    paymentUrl: "https://" + host + "/eng/process",
    fields: clean,
  });
});

module.exports = router;

Frontend — submit the payment form (React)

PayFast requires a real HTML form POST to their server. You cannot use fetch() or a redirect. Build the form in JavaScript and submit it.

// components/PayButton.jsx
import { useState } from "react";

export default function PayButton({ amount, itemName, userId }) {
  const [loading, setLoading] = useState(false);

  async function handlePay() {
    setLoading(true);
    try {
      // Call your backend to get the signed payment data
      const res  = await fetch("/api/create-payment", {
        method:  "POST",
        headers: { "Content-Type": "application/json" },
        body:    JSON.stringify({ amount, itemName, userId }),
      });
      const { paymentUrl, fields } = await res.json();

      // Build a hidden form and submit it to PayFast
      const form    = document.createElement("form");
      form.method   = "POST";
      form.action   = paymentUrl;

      Object.entries(fields).forEach(([key, value]) => {
        const input   = document.createElement("input");
        input.type    = "hidden";
        input.name    = key;
        input.value   = value;
        form.appendChild(input);
      });

      document.body.appendChild(form);
      form.submit(); // User is taken to PayFast's hosted payment page

    } catch (err) {
      console.error(err);
      setLoading(false);
    }
  }

  return (
    <button onClick={handlePay} disabled={loading}>
      {loading ? "Redirecting to PayFast..." : `Pay R${amount}`}
    </button>
  );
}

Handling subscription events in your ITN webhook

function handleITN(pfData) {
  const {
    payment_status,   // "COMPLETE", "FAILED", "CANCELLED"
    token,            // PayFast's subscription token — SAVE THIS, you need it to cancel
    custom_str1,      // your userId
    amount_gross,     // amount actually charged
    billing_date,     // next billing date
  } = pfData;

  if (payment_status === "COMPLETE") {
    db.subscriptions.upsert({
      userId:   custom_str1,
      token:    token,        // ← critical — save this to your database
      status:   "active",
      nextBill: billing_date,
    });
  }

  if (payment_status === "FAILED") {
    emailService.notifyPaymentFailed(custom_str1);
  }

  if (payment_status === "CANCELLED") {
    db.subscriptions.update({ userId: custom_str1, status: "cancelled" });
  }
}

  const host = process.env.PAYFAST_SANDBOX === "true"

  const res = await fetch(
    "https://" + host + "/eng/recurring/cancel/" + token,
    { method: "PUT", headers: buildApiHeaders() }
  );
  return res.ok;
}
Critical: Send HTTP 200 as the very first thing you do. PayFast will retry the notification multiple times if you don't respond fast enough. Process the data AFTER sending the response.

Complete ITN handler (Node.js / Express)

// routes/notify.js
const express = require("express");
const router  = express.Router();
const { validateITN } = require("../utils/payfast");

router.post("/payfast/notify",
  express.urlencoded({ extended: false }),  // PayFast sends form-encoded data
  async (req, res) => {

  // STEP 1: Respond 200 immediately — do this before anything else
  res.status(200).send("OK");

  const pfData  = req.body;
  const rawBody = new URLSearchParams(req.body).toString();

  try {
    // STEP 2: Validate the ITN (checks signature + pings PayFast)
    const { valid, reason } = await validateITN(pfData, rawBody);
    if (!valid) {
      console.error("Rejected ITN:", reason, pfData);
      return;
    }

    // STEP 3: Cross-check the amount to catch fraud
    const expected = await db.orders.getExpectedAmount(pfData.m_payment_id);
    const received = parseFloat(pfData.amount_gross);
    if (Math.abs(received - expected) > 0.01) {
      console.error("Amount mismatch! Expected:", expected, "Got:", received);
      return;
    }

    // STEP 4: Act on the payment status
    switch (pfData.payment_status) {
      case "COMPLETE":
        await db.orders.markPaid(pfData.m_payment_id);
        await db.users.grantAccess(pfData.custom_str1);

        // If it has a token, it's a subscription payment
        if (pfData.token) {
          await db.subscriptions.activate(pfData.custom_str1, pfData.token);
        }
        // If it has an affiliate code, queue a commission
        if (pfData.custom_str2) {
          await queueAffiliateCommission(pfData);
        }
        break;

      case "FAILED":
        await emailService.notifyFailed(pfData.custom_str1);
        break;

      case "PENDING":
        // EFT payments start as PENDING — just wait for COMPLETE
        console.log("EFT payment pending:", pfData.m_payment_id);
        break;
    }

  } catch (err) {
    console.error("ITN processing error:", err);
    // Don't rethrow — we already sent the 200
  }
});

module.exports = router;

Track affiliates using custom fields

PayFast echoes custom_str1 through custom_str5 back in the ITN notification. Store your affiliate code there and act on it when payment completes.

// When someone arrives via an affiliate link: yoursite.com/?ref=AFF123
// Save the code in a cookie, then attach it to the payment:

const data = {
  ...basePaymentFields,
  custom_str1: String(userId),
  custom_str2: req.cookies.affiliateCode  || "",   // "AFF123"
  custom_str3: req.cookies.campaignName   || "",
};

// In your ITN handler, when payment_status === "COMPLETE":
async function queueAffiliateCommission(pfData) {
  const affiliateCode = pfData.custom_str2;
  if (!affiliateCode) return;

  const affiliate = await db.affiliates.findByCode(affiliateCode);
  if (!affiliate) return;

  const saleAmount  = parseFloat(pfData.amount_gross);
  const commission  = saleAmount * (affiliate.commissionRate || 0.15); // 15%

  await db.commissions.create({
    affiliateId:  affiliate.id,
    orderId:      pfData.m_payment_id,
    saleAmount,
    commission,
    status: "pending",    // becomes "payable" after your hold period (e.g. 30 days)
  });

  console.log("Commission queued: R" + commission.toFixed(2) + " for " + affiliateCode);
}

Set up campaigns with a list of tasks (sign up, test a feature, submit feedback). Track user progress. When all tasks are done, the reward fires automatically.

Database schema (PostgreSQL)

-- A campaign = a set of tasks with a total reward
CREATE TABLE campaigns (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title       TEXT NOT NULL,
  description TEXT,
  reward_zar  DECIMAL(10, 2) NOT NULL,
  is_active   BOOLEAN DEFAULT true,
  created_at  TIMESTAMP DEFAULT NOW()
);

-- Individual tasks within a campaign
CREATE TABLE tasks (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  campaign_id UUID REFERENCES campaigns(id) ON DELETE CASCADE,
  title       TEXT NOT NULL,
  description TEXT,
  task_url    TEXT,        -- the URL the user must visit or sign up on
  sort_order  INTEGER DEFAULT 0
);

-- Track which users completed which tasks
CREATE TABLE completions (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id      UUID NOT NULL,
  task_id      UUID REFERENCES tasks(id),
  campaign_id  UUID REFERENCES campaigns(id),
  verified     BOOLEAN DEFAULT true,
  completed_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id, task_id)   -- prevents duplicate completions
);

-- Payout queue — one row per user per campaign
CREATE TABLE reward_payouts (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     UUID NOT NULL,
  campaign_id UUID REFERENCES campaigns(id),
  amount_zar  DECIMAL(10, 2) NOT NULL,
  status      TEXT DEFAULT 'pending',   -- pending | processing | paid | failed
  payout_ref  TEXT,
  created_at  TIMESTAMP DEFAULT NOW(),
  paid_at     TIMESTAMP,
  UNIQUE(user_id, campaign_id)   -- prevents paying the same person twice
);

API — complete a task and trigger the reward

// POST /api/tasks/:taskId/complete
router.post("/tasks/:taskId/complete", requireAuth, async (req, res) => {
  const userId = req.user.id;
  const task   = await db.tasks.findById(req.params.taskId);

  if (!task) return res.status(404).json({ error: "Task not found" });

  // Record the completion (UNIQUE constraint prevents duplicates)
  await db.completions.upsert({
    userId,
    taskId:     task.id,
    campaignId: task.campaign_id,
    verified:   true,
  });

  // Count total tasks vs completed tasks for this campaign
  const total     = await db.tasks.count({ where: { campaign_id: task.campaign_id } });
  const completed = await db.completions.count({
    where: { user_id: userId, campaign_id: task.campaign_id, verified: true },
  });

  // If all done, trigger the reward
  if (completed >= total) {
    await triggerReward(userId, task.campaign_id);
  }

  res.json({ completed, total, allDone: completed >= total });
});

async function triggerReward(userId, campaignId) {
  const campaign = await db.campaigns.findById(campaignId);
  const user     = await db.users.findById(userId);

  // Create the payout record (UNIQUE constraint prevents double-paying)
  const payout = await db.rewardPayouts.create({
    userId,
    campaignId,
    amountZar: campaign.reward_zar,
    status: "pending",
  }).catch(() => null);  // null = already exists, skip

  if (!payout) return;

  // Send the money via Ozow Payouts API
  if (user.bankAccount) {
    await ozowPayout({
      bankAccount: user.bankAccount,
      branchCode:  user.branchCode,
      amount:      campaign.reward_zar,
      reference:   "reward_" + payout.id,
      name:        user.fullName,
    });
    await db.rewardPayouts.update(payout.id, { status: "processing" });
  } else {
    // No bank account on file — email the user to claim their reward
    await emailService.sendRewardNotification(user.email, {
      amount:   campaign.reward_zar,
      campaign: campaign.title,
    });
  }
}

How it works

PayPal Payouts is completely separate from the normal PayPal checkout flow. It is a server-to-server API where you push money out to your affiliates. No invoice, no request — you just send it. All you need is their PayPal email address.

Step 1 — Enable Payouts on your PayPal account

1. Log into developer.paypal.com
2. Go to your app → under "Features" enable "Payouts"
   (you may need to request this from PayPal support if not visible)
3. Get your Client ID and Client Secret from the app dashboard
4. Add them to your .env file:
PAYPAL_CLIENT_ID=your_client_id_here
PAYPAL_CLIENT_SECRET=your_client_secret_here
PAYPAL_SANDBOX=true   # set to false when going live

# Sandbox API:  https://api-m.sandbox.paypal.com
# Live API:     https://api-m.paypal.com

Step 2 — utils/paypal.js — authentication helper

PayPal uses OAuth2. You get a token first, then use it for all API calls. Tokens last 9 hours so cache them.

// utils/paypal.js

let cachedToken     = null;
let tokenExpiresAt  = 0;

async function getPayPalToken() {
  // Return cached token if still valid
  if (cachedToken && Date.now() < tokenExpiresAt) {
    return cachedToken;
  }

  const base = process.env.PAYPAL_SANDBOX === "true"
    ? "https://api-m.sandbox.paypal.com"
    : "https://api-m.paypal.com";

  const credentials = Buffer.from(
    process.env.PAYPAL_CLIENT_ID + ":" + process.env.PAYPAL_CLIENT_SECRET
  ).toString("base64");

  const res = await fetch(base + "/v1/oauth2/token", {
    method:  "POST",
    headers: {
      "Authorization": "Basic " + credentials,
      "Content-Type":  "application/x-www-form-urlencoded",
    },
    body: "grant_type=client_credentials",
  });

  const data         = await res.json();
  cachedToken        = data.access_token;
  tokenExpiresAt     = Date.now() + (data.expires_in - 60) * 1000; // 60s buffer

  return cachedToken;
}

module.exports = { getPayPalToken };

Step 3 — Send a single payout to one affiliate

// utils/paypalPayout.js
const { getPayPalToken } = require("./paypal");

/**
 * Pay one affiliate via PayPal.
 *
 * recipientEmail  = their PayPal email address
 * amount          = number e.g. 25.00
 * currency        = "USD", "EUR", "GBP", "ZAR" etc.
 * note            = message the recipient sees
 * senderItemId    = your internal reference (e.g. commission ID)
 */
async function sendPayPalPayout({ recipientEmail, amount, currency = "USD", note, senderItemId }) {
  const base  = process.env.PAYPAL_SANDBOX === "true"
    ? "https://api-m.sandbox.paypal.com"
    : "https://api-m.paypal.com";

  const token = await getPayPalToken();

  const body = {
    sender_batch_header: {
      sender_batch_id: "batch_" + Date.now(),   // must be unique per batch
      email_subject:   "You have a payment from us!",
      email_message:   "Your affiliate commission has been paid.",
    },
    items: [
      {
        recipient_type: "EMAIL",
        receiver:       recipientEmail,
        amount: {
          value:    amount.toFixed(2),
          currency: currency,
        },
        note:           note || "Affiliate commission",
        sender_item_id: String(senderItemId),
      },
    ],
  };

  const res = await fetch(base + "/v1/payments/payouts", {
    method:  "POST",
    headers: {
      "Authorization": "Bearer " + token,
      "Content-Type":  "application/json",
    },
    body: JSON.stringify(body),
  });

  const result = await res.json();

  if (!res.ok) {
    throw new Error("PayPal Payout failed: " + JSON.stringify(result));
  }

  // result.batch_header.payout_batch_id = PayPal's reference for this payout
  return result;
}

module.exports = { sendPayPalPayout };

Step 4 — Batch payout (pay many affiliates at once)

More efficient than sending one at a time. PayPal processes the whole batch and handles each recipient individually.

async function sendBatchPayouts(commissions) {
  // commissions = array of { email, amount, currency, commissionId }

  const base  = process.env.PAYPAL_SANDBOX === "true"
    ? "https://api-m.sandbox.paypal.com"
    : "https://api-m.paypal.com";

  const token = await getPayPalToken();

  const items = commissions.map((c, index) => ({
    recipient_type: "EMAIL",
    receiver:       c.email,
    amount: {
      value:    c.amount.toFixed(2),
      currency: c.currency || "USD",
    },
    note:           "Affiliate commission payment",
    sender_item_id: String(c.commissionId),
  }));

  const res = await fetch(base + "/v1/payments/payouts", {
    method:  "POST",
    headers: {
      "Authorization": "Bearer " + token,
      "Content-Type":  "application/json",
    },
    body: JSON.stringify({
      sender_batch_header: {
        sender_batch_id: "batch_" + Date.now(),
        email_subject:   "Your affiliate commission is here",
      },
      items,
    }),
  });

  return res.json();
}

// Example: run every Monday to pay the previous week's commissions
async function weeklyPayoutJob() {
  const pending = await db.commissions.findAll({
    where: { status: "payable" },   // "payable" = past your hold period
    include: ["affiliate"],
  });

  if (pending.length === 0) return;

  const result = await sendBatchPayouts(
    pending.map(c => ({
      email:        c.affiliate.paypalEmail,
      amount:       c.commission,
      currency:     c.affiliate.currency || "USD",
      commissionId: c.id,
    }))
  );

  // Mark them as paid
  await db.commissions.updateMany(
    { id: { in: pending.map(c => c.id) } },
    { status: "paid", paidAt: new Date(), payoutBatchId: result.batch_header.payout_batch_id }
  );

  console.log("Batch payout sent:", result.batch_header.payout_batch_id);
}

// Check the status of a batch payout
async function getPayoutStatus(payoutBatchId) {
  const base  = process.env.PAYPAL_SANDBOX === "true"
    ? "https://api-m.sandbox.paypal.com"
    : "https://api-m.paypal.com";

  const token = await getPayPalToken();

  const res = await fetch(base + "/v1/payments/payouts/" + payoutBatchId, {
    headers: { "Authorization": "Bearer " + token },
  });

  const data = await res.json();

  // data.batch_header.batch_status = "PENDING" | "PROCESSING" | "SUCCESS" | "DENIED" | "CANCELED"
  // data.items = array of individual payout items with their own status
  return data;
}

// Individual item statuses:
// SUCCESS     = money delivered to recipient
// FAILED      = delivery failed (bad email, unclaimed, etc.)
// UNCLAIMED   = recipient has no PayPal account — they get an email to claim it
// RETURNED    = unclaimed for 30 days, money returned to you
// BLOCKED     = PayPal blocked it (compliance etc.)

The complete routing logic — SA vs international

// This is the single function you call for ALL affiliate payouts.
// It automatically routes to the right provider.

const { sendPayPalPayout }  = require("./utils/paypalPayout");
const { ozowPayout }        = require("./utils/ozow");

async function payAffiliate(affiliate, commission) {
  const isSouthAfrican = affiliate.country === "ZA";

  if (isSouthAfrican && affiliate.bankAccount) {
    // SA affiliate with bank account → Ozow direct bank transfer
    await ozowPayout({
      bankAccount: affiliate.bankAccount,
      branchCode:  affiliate.branchCode,
      amount:      commission.amount,
      reference:   "comm_" + commission.id,
      name:        affiliate.fullName,
    });
    await db.commissions.update(commission.id, { status: "paid", method: "ozow" });

  } else if (affiliate.paypalEmail) {
    // Everyone else (or SA affiliates who prefer PayPal) → PayPal Payouts
    await sendPayPalPayout({
      recipientEmail: affiliate.paypalEmail,
      amount:         commission.amount,
      currency:       affiliate.currency || "USD",
      note:           "Commission for order " + commission.orderId,
      senderItemId:   commission.id,
    });
    await db.commissions.update(commission.id, { status: "paid", method: "paypal" });

  } else {
    // No payout method on file — notify them to add one
    await emailService.sendAddPayoutMethodEmail(affiliate.email, commission.amount);
    await db.commissions.update(commission.id, { status: "on_hold", reason: "no_payout_method" });
  }
}
